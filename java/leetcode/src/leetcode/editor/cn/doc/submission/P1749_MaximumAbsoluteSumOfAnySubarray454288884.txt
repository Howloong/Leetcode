    class Solution {
        public int maxAbsoluteSum(int[] nums) {
            int max = 0;
            int sum = 0;
            int min = 0;
            for (int num : nums) {
                sum += num;
                if (sum > max) {
                    max = sum;
                } else if (sum < min) {
                    min = sum;
                }
            }
            return max-min;
        }
    }
