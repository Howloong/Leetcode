    class Solution {
        public void reorderList(ListNode head) {
            if (head == null) {
                return;
            }
            Deque<ListNode> deque = new ArrayDeque<>();
            while (head != null) {
                deque.push(head);
                head = head.next;
            }
            head = new ListNode();
            ListNode res = head;
            while (!deque.isEmpty()) {
                head.next = deque.pollLast();
                head = head.next;
                head.next = null;
                if (!deque.isEmpty()) {
                    head.next = deque.pollFirst();
                    head = head.next;
                    head.next = null;
                }
            }
            head = res.next;
        }
    }
