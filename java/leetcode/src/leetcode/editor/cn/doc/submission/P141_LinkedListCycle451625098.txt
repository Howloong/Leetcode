public class Solution {
    public boolean hasCycle(ListNode head) {
        ListNode p = head;
        ListNode q = head;
        while (q != null) {
            p = p.next;
            if (q.next == null || q.next.next == null) {
                return false;
            }
            q = q.next.next;
            if (q == p) {
                return true;
            }
        }
        return false;
    }
}
