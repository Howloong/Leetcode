    class Solution {
        public int findPoisonedDuration(int[] timeSeries, int duration) {
            int sum = 0;
            int last = timeSeries[0];
            for (int i = 1; i < timeSeries.length; i++) {
                if (timeSeries[i] - last > duration - 1) {
                    sum += duration;
                } else {
                    sum += (timeSeries[i] - last);
                }
                last = timeSeries[i];
            }
            sum += duration;
            return sum;
        }
    }
